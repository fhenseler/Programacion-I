/**
 * \brief Utiliza la función "add" para sumar dos números e imprime el resultado
 * \param double num1 Primer número a sumar
 * \param double num2 Segundo número a sumar
 * \return void
 */
void printAdd(double num1, double num2)
{
    double result = add(num1, num2);
    printf("\nSUMA ( A + B ) = %.5lf", result);
}


/**
 * \brief Utiliza la función "sub" para restar dos números e imprime el resultado
 * \param double num1 Primer número a restar
 * \param double num2 Segundo número a restar
 * \return void
 */
void printSub(double num1, double num2)
{
    double result = sub(num1, num2);
    printf("\nRESTA ( A - B ) = %.5lf", result);
}


/**
 * \brief Utiliza la función "divi" para dividir dos números e imprime el resultado
 * \param double num1 Primer número a dividir
 * \param double num2 Segundo número a dividir
 * \return void
 */
void printDivi(double num1, double num2)
{
    double result = divi(num1, num2);
    printf("\nDIVISI%cN ( A / B ) = %.5lf", 224, result);
}


/**
 * \brief Utiliza la función "mult" para multiplicar dos números e imprime el resultado
 * \param double num1 Primer número a multiplicar
 * \param double num2 Segundo número a multiplicar
 * \return void
 */
void printMult(double num1, double num2)
{
    double result = mult(num1, num2);
    printf("\nMULTIPLICACI%cN ( A * B ) = %.5lf", 224, result);
}


/**
 * \brief Utiliza la función "factorial" para calcular el factorial de un número e imprime el resultado
 * \param double num1 Número al que se le calculará el factorial
 * \return void
 */
void printFactorial(double num1)
{
    unsigned long result = factorial(num1);
    printf("\nFACTORIAL ( A !) = %lu", result);
}
